༄༅། །རྒྱུན་ཁྱེར་གྱི་བསངས་ཡིག་བསྡུས་པ་བཞུགས་སོ།། ཨོཾ་སྭ་སྟི། བསངས་རྫས་ཅི་འབྱོར་བ་ཆབ་ཀྱིས་བྲན་ཏེ། སངས་རྒྱས་ཆོས་ཚོགས་མ་ཚར་གསུམ་གྱིས་སྐྱབས་སེམས་བྱས་ལ། ཨོཾ་ཆོས་ཉིད་མི་འགྱུར་རྟེན་ཅིང་འབྲེལ་འབྱུང་དང་། །དཀོན་མཆོག་ཐུགས་རྗེས་བདག་གི་མོས་པ་ཡིས། །བསངས་མཆོད་སྣ་ཚོགས་བཤམས་ཤིང་སྤྲུལ་པ་རྣམས། །མཁའ་དབྱིངས་ཁྱབ་པ་དུས་ཀུན་འབྱུང་བ་དག །རང་རང་མོས་པ་བཞིན་དུ་སྤྱོད་གྱུར་ཅིག །ཉམས་གྲིབ་སེལ་བའི་དྲི་ཞིམ་བསངས་མཆོད་འདིས། །ལུང་རྟོགས་བསྟན་པ་གང་ལ་རག་ལས་པས། །རྩ་བརྒྱུད་འདྲེན་པའི་བླ་མ་ཐམས་ཅད་བསངས། །ཡི་དམ་ལྷ་ཚོགས་སངས་རྒྱས་བྱང་སེམས་བསངས། །བསྟན་སྲུང་ནོར་ལྷ་དཀར་ཕྱོགས་ཐམས་ཅད་བསངས། །ཁྱད་པར་ཡུལ་འདིར་གནས་པའི་ཡུལ་ལྷ་བསངས། །རང་དང་གྲོགས་ལ་འགོ་བའི་ལྷ་རྣམས་བསངས། །མགོན་སྐྱབས་བྱེད་དང་ལམ་ལ་གནས་པ་བསངས། །མདོར་ན་འདིར་མང་སྨོས་པས་ཅི་ཞིག་འཚལ། །སྣང་སྲིད་སྐྱེ་འགྲོ་ཡོད་དོ་ཅོག་ཀུན་བསངས། །དེ་ལྟར་བསངས་མཆོད་ཕུལ་བའི་བྱིན་གྱིས་མཐུས། །བདག་ཅག་བསམ་དོན་ཆོས་བཞིན་འགྲུབ་པ་དང་། །མི་ལ་ན་ཚ་ཕྱུགས་ལ་གོད་མ་བཏང་། །ལེགས་པའི་གཞི་ཐིངས་ཉེས་པའི་ཕྱོགས་ལས་སྲུངས། །རྐྱེན་དབང་ཕན་ཚུན་འཛེར་འཕོག་བྱུང་བ་རྣམས། །མཁའ་ལ་སྤྲིན་དེངས་བཞིན་དུ་བཟོད་པར་གསོལ། །རྩ་བརྒྱུད་བླ་མ་མཆོག་གསུམ་ཐུགས་རྗེ་དང་། །འདིར་མཚོན་རྣམ་དཀར་དགེ་ཚོགས་ཀུན་གྱི་མཐུས། །ཚོགས་སྤེལ་བགེགས་སེལ་འདོད་དོན་ཀུན་འགྲུབ་པའི། །བཀྲ་ཤིས་དམ་པས་དེང་འདིར་བདེ་ལེགས་ཤོག །རྒྱུན་ཁྱེར་གྱིས་བསངས་ཡིག་བསྡུས་པ་འདིའང་སངས་རྒྱས་ཕུན་ཚོགས་ཀྱིས་སོ།། །།སརྦ་མངྒ་ལཾ། །དགེའོ་ཞལ་གྲོ། །། PAGE \* MERGEFORMAT 2 ཀླུ་སྒྲུབ། རྒྱུན་ཁྱེར་གྱི་བསངས་ཡིག་བསྡུས་པ་ལ། PAGE \* MERGEFORMAT 1
